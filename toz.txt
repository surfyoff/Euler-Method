#include <iostream>
using namespace std;

// Calculate Acceleration
double accelCalc(double A,double B,double Speed) {
	return A - B * Speed;
}

void EulerMethod(double A, double B, double initSpeed, double timeStep, int numSteps) {
	double currentSpeed = initSpeed;
	double currentAcceleration = accelCalc(A, B, initSpeed);
	for (int i = 0; i < numSteps+1; i++) {
		// Current Values of Speed and Acceleration for a specific Step
		cout << "Step " << i << " : " << endl;
		cout << " Speed" << i << " = " << currentSpeed << endl;
		cout << " Acceleration" << i << " = " << currentAcceleration << endl << endl;	

		// Calculate the next Speed and Acceleration
		double nextSpeed = currentSpeed + currentAcceleration * timeStep;
		double nextAcceleration = accelCalc(A, B, nextSpeed);

		// Update Speed and Acceleration values for the next iteration
		currentSpeed = nextSpeed;
		currentAcceleration = nextAcceleration;
	}

}
int main() {
	double A, B, initSpeed, timeStep;
	int n, numSteps;
	cout << "Please input these informations :\n";
	cout << "Initial Speed : "; cin >> initSpeed;
	cout << "Time step : "; cin >> timeStep;
	cout << "A : "; cin >> A;
	cout << "B : "; cin >> B;
	cout << "n : "; cin >> n;
	cout << "numSteps : "; cin >> numSteps;

	EulerMethod(A, B, initSpeed, timeStep, numSteps);
}
